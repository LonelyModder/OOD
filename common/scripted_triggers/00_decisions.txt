add_scaling_gold_cost_trigger = {
	trigger_if = {
		limit = {
			scope:player = {
				treasury < root.calc_governor_investment_fee
				OR = {
					NOT = {
						has_variable = free_investments
					}
					var:free_investments <= 0
				}
			}
		}
		scope:player = {
			treasury >= root.calc_governor_investment_fee
		}
	}
}
