ESFA_on_create_character_effect = {
	if = {
		limit = {
			NOT = { has_trait = annunaki }
			NOT = { has_trait = igigi }
			NOT = { has_trait = ekimmu }
			NOT = { has_trait = rabisu }
			NOT = { has_trait = human }
		}
		if = {
			limit = {
				OR = {
					has_culture_group = annunaki
				}
			}
			add_trait = annunaki
		}
		else_if = {
			limit = {
				has_culture_group = igigi
			}
			add_trait = igigi
		}
		else_if = {
			limit = {
				has_culture_group = ekimmu
			}
			add_trait = ekimmu
		}
		else_if = {
			limit = {
				has_culture_group = rabisu
			}
			add_trait = rabisu
		}
		else = {
			add_trait = human
		}
	}
}
