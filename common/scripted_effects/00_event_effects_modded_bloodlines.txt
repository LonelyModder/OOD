
pass_down_traits_effect = {
	scope:newborn = {
		if = {
			limit = {
				has_trait = human
			}
			if = {
				limit = {
					exists = father
					father = {
						OR = {
							has_trait = bloodofscorpion
							has_trait = urur
							has_trait = anittas
						}
					}
							
				}
				if = {
					limit = {
						father = {
							has_trait = bloodofscorpion
						}
						NOT = {
							has_trait = bloodofscorpion
						}
					}
					add_trait = bloodofscorpion
				}
				if = {
					limit = {
						father = {
							has_trait = urur
						}
						NOT = {
							has_trait = urur
						}
					}
					add_trait = urur
				}
				if = {
					limit = {
						father = {
							has_trait = anittas
						}
						NOT = {
							has_trait = anittas
						}
					}
					add_trait = anittas
				}
			}
			if = {
				limit = {
					exists = mother
					mother = {
						has_trait = bloodofscorpion
					}
					NOT = {
						has_trait = bloodofscorpion
					}
				}
				add_trait = bloodofscorpion
			}
			if = {
				limit = {
					exists = mother
					mother = {
						has_trait = urur
					}
					NOT = {
						has_trait = urur
					}
				}
				add_trait = urur
			}
			if = {
				limit = {
					exists = mother
					mother = {
						has_trait = anittas
					}
					NOT = {
						has_trait = anittas
					}
				}
				add_trait = anittas
			}
		}
	}
}

bastard_major_bloodline_effect = {
	if = {
		limit = {
			has_trait = human
		}
		if = {
			limit = {
				scope:bastard_ruler = {
					has_trait = bloodofscorpion
				}
			}
			add_trait = bloodofscorpion
		}
		if = {
			limit = {
				scope:bastard_ruler = {
					has_trait = urur
				}
			}
			add_trait = urur
		}
		if = {
			limit = {
				scope:bastard_ruler = {
					has_trait = anittas
				}
			}
			add_trait = anittas
		}
	}
}
