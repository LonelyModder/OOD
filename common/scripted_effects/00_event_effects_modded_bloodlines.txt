
pass_down_traits_effect = {
	scope:newborn = {
		if = {
			limit = {
				exists = father
				father = {
					OR = {
						has_trait = bloodofscorpion
					}
				}
						
			}
			if = {
				limit = {
					father = {
						has_trait = bloodofscorpion
					}
					NOT = {
						has_trait = bloodofscorpion
					}
				}
				add_trait = bloodofscorpion
			}
		}
		if = {
			limit = {
				exists = mother
				mother = {
					has_trait = bloodofscorpion
				}
				NOT = {
					has_trait = bloodofscorpion
				}
			}
			add_trait = bloodofscorpion
		}
	}
}

bastard_major_bloodline_effect = {
	if = {
		limit = {
			scope:bastard_ruler = {
				has_trait = bloodofscorpion
			}
		}
		add_trait = bloodofscorpion
	}
}
