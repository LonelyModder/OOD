namespace = bloodlines_startup

bloodlines_startup.1 = { #Salgari

	type = country_event
	hidden = yes
	
	trigger = {
		OR = {
			tag = ABK
			tag = ABI
		}
	}
	
	immediate = {
		every_character = {
			limit = {
				OR = {
					has_same_family = char:0
					has_same_family = char:3
				}
			}
			add_trait = bloodofscorpion
		
		}
		hidden_effect = {
			trigger_event = { id = bloodlines_setup_completion.1 }
		}
	
	}

}
