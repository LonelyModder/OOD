namespace = bloodlines_setup_completion

bloodlines_setup_completion.1 = {
	type = country_event
	hidden = yes
	
	immediate = {
		every_character = {
			limit = {
				is_alive = yes
				OR = {
					exists = father
					exists = mother
				}
			}
			if = {
				limit = {
					has_trait = human
				}
				if = {
					limit = {
						exists = father
						father = {
							has_trait = bloodofscorpion
						}
						NOT = {
							has_trait = bloodofscorpion
						}
					}
					add_trait = bloodofscorpion
				}
				if = {
					limit = {
						exists = father
						father = {
							has_trait = urur
						}
						NOT = {
							has_trait = urur
						}
					}
					add_trait = urur
				}
				if = {
					limit = {
						exists = father
						father = {
							has_trait = anittas
						}
						NOT = {
							has_trait = anittas
						}
					}
					add_trait = anittas
				}
				if = {
					limit = {
						exists = mother
						mother = {
							has_trait = bloodofscorpion
						}
						NOT = {
							has_trait = bloodofscorpion
						}
					}
					add_trait = bloodofscorpion	
				}
				if = {
					limit = {
						exists = mother
						mother = {
							has_trait = urur
						}
						NOT = {
							has_trait = urur
						}
					}
					add_trait = urur	
				}
				if = {
					limit = {
						exists = mother
						mother = {
							has_trait = anittas
						}
						NOT = {
							has_trait = anittas
						}
					}
					add_trait = anittas	
				}
			}
		}
	}
}
